import type { Express, Request, Response, NextFunction } from "express";
import { createServer, type Server } from "http";
import { storage } from "./storage";
import { emailService } from "./email";
import { ImapService } from "./imap-service";
import { generateComparison } from "./routes/compare";
import { sendFollowUp } from "./routes/follow-up";
import { downloadAttachment } from "./routes/attachments";
import { registerUser, loginUser, logoutUser, getCurrentUser, authMiddleware } from "./auth";
import { 
  searchRequestFormSchema, 
  emailTemplateSchema,
  insertSupplierResponseSchema,
  insertUserSchema,
  userLoginSchema,
  insertAnalysisResultSchema,
  insertContactGroupSchema,
  insertContactItemSchema,
  insertRequestSupplierGroupSchema,
  insertEmailRequestSchema,
  type SearchRequest,
  type InsertSearchRequest,
  type RequestSupplier,
  type SupplierResponse,
  type Supplier,
  type AnalysisResult,
  type User,
  type ContactGroup,
  type InsertContactGroup,
  type ContactItem,
  type InsertContactItem,
  type EmailRequest,
  type InsertEmailRequest,
  type RequestSupplierGroup,
  type InsertRequestSupplierGroup
} from "@shared/schema";

export async function registerRoutes(app: Express): Promise<Server> {
  // Тестовый маршрут для проверки отправки email
  app.get("/api/test-email", async (req, res) => {
    console.log("Testing email sending...");
    
    try {
      const testSuccess = await emailService.sendEmail({
        to: "test@example.com",
        subject: "Test Email",
        text: "This is a test email to verify the email sending functionality",
        headers: {
          'X-Test': 'Test Header'
        }
      });
      
      console.log(`Test email sending: ${testSuccess ? "Successful" : "Failed"}`);
      
      res.json({ 
        success: testSuccess,
        message: testSuccess ? "Test email sent successfully" : "Failed to send test email"
      });
    } catch (error) {
      console.error("Error in test email route:", error);
      res.status(500).json({ success: false, error: String(error) });
    }
  });
  // Authentication routes
  app.post("/api/register", registerUser);
  app.post("/api/login", loginUser);
  app.post("/api/logout", logoutUser);
  app.get("/api/user", getCurrentUser);
  
  // Get all suppliers
  app.get("/api/suppliers", async (req, res) => {
    try {
      const searchParam = req.query.search as string;
      let suppliers = await storage.getSuppliers();
      
      // If search parameter is provided, filter suppliers
      if (searchParam) {
        const search = searchParam.toLowerCase();
        suppliers = suppliers.filter(supplier => 
          supplier.name.toLowerCase().includes(search) || 
          supplier.email.toLowerCase().includes(search) ||
          (supplier.description && supplier.description.toLowerCase().includes(search))
        );
      }
      
      res.json(suppliers);
    } catch (error) {
      console.error("Error fetching suppliers:", error);
      res.status(500).json({ message: "Failed to fetch suppliers", error: String(error) });
    }
  });
  
  // Get groups for a supplier
  app.get("/api/supplier-groups/:supplierId", async (req, res) => {
    try {
      const supplierId = parseInt(req.params.supplierId);
      if (isNaN(supplierId)) {
        return res.status(400).json({ message: "Invalid supplier ID" });
      }
      
      const groupIds = await storage.getSupplierGroups(supplierId);
      res.json(groupIds);
    } catch (error) {
      console.error(`Error fetching groups for supplier ${req.params.supplierId}:`, error);
      res.status(500).json({ message: "Failed to fetch supplier groups", error: String(error) });
    }
  });
  
  // Update groups for a supplier
  app.post("/api/suppliers/:supplierId/groups", async (req, res) => {
    try {
      const supplierId = parseInt(req.params.supplierId);
      if (isNaN(supplierId)) {
        return res.status(400).json({ message: "Invalid supplier ID" });
      }
      
      const { groupIds } = req.body;
      if (!Array.isArray(groupIds)) {
        return res.status(400).json({ message: "groupIds must be an array of numbers" });
      }
      
      const result = await storage.updateSupplierGroups(supplierId, groupIds);
      
      if (result.success) {
        res.json({ success: true, message: "Supplier groups updated successfully" });
      } else {
        res.status(400).json({ success: false, message: "Failed to update supplier groups" });
      }
    } catch (error) {
      console.error(`Error updating groups for supplier ${req.params.supplierId}:`, error);
      res.status(500).json({ message: "Failed to update supplier groups", error: String(error) });
    }
  });

  // Create search request and get matched suppliers
  app.post("/api/search-requests", async (req, res) => {
    try {
      // Log the incoming request to debug
      console.log("Request body:", req.body);

      // Validate the data with the simplified schema
      const validatedData = searchRequestFormSchema.parse(req.body);

      // Check for duplicate requests in the last 30 seconds
      const recentRequests = await storage.getPendingSearchRequests();
      const duplicateRequest = recentRequests.find(req => {
        // Safely handle null/undefined createdAt dates
        const createdAtTime = req.createdAt ? 
          (req.createdAt instanceof Date ? req.createdAt : new Date(req.createdAt)).getTime() : 0;
        const isRecent = Date.now() - createdAtTime < 30000;

        return isRecent && 
               req.productName === validatedData.productName && 
               req.productDescription === validatedData.productDescription;
      });

      if (duplicateRequest) {
        // Use the original request with its original order number
        return res.json({ 
          request: duplicateRequest,
          matchedSuppliers: await storage.searchSuppliers(duplicateRequest)
        });
      }

      // Create search request - the storage layer will generate the order number
      const searchRequestData = {
        productName: validatedData.productName,
        productDescription: validatedData.productDescription,
        timeline: validatedData.timeline,
        quantity: validatedData.quantity,
        budget: validatedData.budget,
        useDbSearch: validatedData.useDbSearch ?? true,
        useApiSearch: validatedData.useApiSearch ?? false,
        orderNumber: '', // This placeholder will be replaced by the generateOrderNumber() method
        // userId temporarily disabled until database schema is updated
        // userId: req.session.userId || null
      };

      // Create search request
      const request = await storage.createSearchRequest(searchRequestData);
      console.log("Created search request:", request);

      // Convert request to plain object if needed (to prevent serialization issues)
      const plainRequest = {
        ...request,
        createdAt: request.createdAt instanceof Date 
          ? request.createdAt 
          : request.createdAt 
            ? new Date(request.createdAt) 
            : new Date()
      };

      // Search for matching suppliers
      const matchedSuppliers = await storage.searchSuppliers(plainRequest);
      console.log("Found matching suppliers:", matchedSuppliers.length);

      // Return the results
      return res.json({ 
        request: {
          ...request,
          // Convert Date to ISO string to avoid serialization issues
          createdAt: request.createdAt instanceof Date 
            ? request.createdAt.toISOString() 
            : request.createdAt
        }, 
        matchedSuppliers 
      });
    } catch (error) {
      console.error("Error processing search request:", error);
      return res.status(400).json({ message: "Invalid request data", error: String(error) });
    }
  });

  // Send email to suppliers
  app.post("/api/send-email", async (req, res) => {
    try {
      // Extract and validate basic fields
      const { suppliers: supplierIds, subject, message, requestId, attachments } = emailTemplateSchema.parse(req.body);

      // Handle search request - either get existing or create a temporary one for direct emails
      let searchRequest: SearchRequest;
      
      if (requestId === 0) {
        // This is a direct email without an associated search request
        // Create a search request in the database to ensure it's tracked
        const orderNumber = storage.generateOrderNumber();
        const tempRequest = {
          productName: subject || 'Прямой запрос',
          productDescription: 'Создано через форму прямого запроса',
          createdAt: new Date(),
          orderNumber: orderNumber,
          status: 'draft',
          timeline: new Date().toISOString(),  // Set to current date
          additionalRequirements: null,
          matchedSuppliers: [],
          useDbSearch: false,
          useApiSearch: false
        };
        
        // Save the temporary request to the database
        searchRequest = await storage.createSearchRequest(tempRequest);
        console.log(`Created and saved search request with order number: ${searchRequest.orderNumber} for direct email. ID: ${searchRequest.id}`);
      } else {
        // Regular case - get the search request from storage
        const existingRequest = await storage.getSearchRequest(requestId);
        if (!existingRequest) {
          return res.status(404).json({ message: `Search request with ID ${requestId} not found` });
        }
        searchRequest = existingRequest;
      }

      // Get all suppliers from database
      const databaseSuppliers = await storage.getSuppliers();
      
      // Проверим, пришли ли ID из групп контактов (не поставщиков)
      const isFromContactGroup = req.body.fromContactGroup === true;
      
      // Handle both database suppliers (positive IDs) and API suppliers (negative IDs)
      let selectedSuppliers = [];

      if (isFromContactGroup) {
        // Получаем контакты из групп по ID
        const contactIds = supplierIds.filter(id => typeof id === 'number' || (typeof id === 'string' && !id.startsWith('api-')));
        console.log("Contact group IDs:", contactIds);
        
        // Получаем детали для всех контактов
        let groupContacts = [];
        for (const contactId of contactIds) {
          try {
            const contact = await storage.getContactById(parseInt(String(contactId)));
            if (contact) {
              console.log(`Found contact: ${contact.name} (ID: ${contact.id}, Email: ${contact.email})`);
              groupContacts.push({
                id: contact.id,
                name: contact.name || `Unknown (${contact.email})`,
                email: contact.email,
                phone: contact.phone || "",
                website: contact.organization || ""
              });
            }
          } catch (error) {
            console.error(`Error getting contact with ID ${contactId}:`, error);
          }
        }
        
        console.log(`Found ${groupContacts.length} contacts from contact groups`);
        selectedSuppliers.push(...groupContacts);
      } else {
        // Стандартная обработка поставщиков
        console.log("Database supplier IDs:", supplierIds.filter(id => typeof id === 'number' || (typeof id === 'string' && !id.startsWith('api-'))));
        
        const dbSelectedSuppliers = databaseSuppliers.filter(s => {
          // Convert both to strings for comparison
          const sIdString = String(s.id);
          const included = supplierIds.some(id => String(id) === sIdString);
          console.log(`Checking DB supplier ${s.name} (ID: ${s.id}): Selected = ${included}`);
          return included;
        });
      
      console.log(`Selected ${dbSelectedSuppliers.length} database suppliers from ${databaseSuppliers.length} available`);
      selectedSuppliers.push(...dbSelectedSuppliers);

      // Process API suppliers if they exist - these will be included directly in the request
      if (req.body.apiSuppliers && Array.isArray(req.body.apiSuppliers)) {
        // Already have full supplier objects from the frontend
        const apiSuppliers = req.body.apiSuppliers;
        
        // Check if suppliers with api- prefix are in the supplierIds array
        const apiPrefixIds = supplierIds.filter(id => typeof id === 'string' && id.startsWith('api-'));
        
        console.log(`API supplier IDs selected (${apiPrefixIds.length}):`, apiPrefixIds);
        
        // Only include API suppliers that were actually selected (with api- prefix in supplierIds)
        const selectedApiSuppliers = apiSuppliers.filter((supplier: Supplier) => {
          // Convert negative ID to positive for string comparison
          const supplierIdStr = String(supplier.id);
          const apiId = `api-${Math.abs(typeof supplier.id === 'number' ? supplier.id : parseInt(supplierIdStr))}`;
          
          // Check both the formatted api-ID and the raw supplier ID string
          const includedByApiId = apiPrefixIds.includes(apiId);
          const includedByRawId = supplierIds.some(id => String(id) === supplierIdStr);
          const included = includedByApiId || includedByRawId;
          
          console.log(`Checking API supplier ${supplier.name} (ID: ${supplier.id}): API ID = ${apiId}, Selected = ${included}`);
          return included;
        });
        
        // Double-check if we found all the suppliers that should be included
        if (selectedApiSuppliers.length !== apiPrefixIds.length) {
          console.warn(`Warning: Found ${selectedApiSuppliers.length} API suppliers, but ${apiPrefixIds.length} were selected. This might indicate a mismatch.`);
          
          // Log the IDs that couldn't be matched
          const foundIds = selectedApiSuppliers.map((s: Supplier) => {
            return `api-${Math.abs(typeof s.id === 'number' ? s.id : parseInt(s.id as string))}`;
          });
          
          const missingIds = apiPrefixIds.filter(id => !foundIds.includes(id));
          if (missingIds.length > 0) {
            console.warn(`Missing supplier IDs: ${missingIds.join(', ')}`);
          }
        }
        
        console.log(`Selected ${selectedApiSuppliers.length} API suppliers from ${apiSuppliers.length} available`);
        selectedSuppliers.push(...selectedApiSuppliers);
      }

      console.log(`Sending emails to ${selectedSuppliers.length} suppliers for request ${searchRequest.orderNumber}`);
      
      // Sanitize and validate attachments
      let sanitizedAttachments: {
        filename: string;
        contentType: string;
        content: string;
        size?: number;
      }[] = [];
      
      if (attachments && Array.isArray(attachments) && attachments.length > 0) {
        console.log(`Request includes ${attachments.length} attachments`);
        
        // Process each attachment to ensure it has valid content
        sanitizedAttachments = attachments.filter(attachment => {
          if (!attachment.filename || !attachment.contentType || !attachment.content) {
            console.warn(`Skipping invalid attachment: missing required fields`);
            return false;
          }
          
          // Check if content is a valid base64 string
          try {
            // Verify it's a base64 string by attempting to decode a small sample
            Buffer.from(attachment.content.substring(0, 10), 'base64');
            return true;
          } catch (err) {
            console.warn(`Skipping attachment with invalid base64 content: ${attachment.filename}`);
            return false;
          }
        });

        console.log(`Processed ${sanitizedAttachments.length} valid attachments out of ${attachments.length}`);
      }

      // Debug log to check the suppliers before sending
      console.log("All selected suppliers:");
      selectedSuppliers.forEach(supplier => {
        console.log(`- ${supplier.name} (${supplier.email})`);
      });

      // Create tracking records and send emails
      const emailResults = await Promise.all(
        selectedSuppliers.map(async (supplier) => {
          try {
            // Generate unique tracking ID for this supplier-request combo
            const trackingId = storage.generateTrackingId();
            
            console.log(`Processing supplier: ${supplier.name} (${supplier.email}) with tracking ID: ${trackingId}`);

            // Create a record of this supplier request
            const requestSupplier = await storage.createRequestSupplier({
              requestId: searchRequest.id,
              supplierId: supplier.id.toString(), // Convert to string for both numeric and string IDs
              supplierEmail: supplier.email,
              supplierName: supplier.name,
              supplierWebsite: supplier.website || '',
              supplierPhone: supplier.phone || '',
              trackingId: trackingId,
              emailSubject: subject,
              emailContent: message,
            });
            
            console.log(`Created requestSupplier record with ID: ${requestSupplier.id}`);

            // Format the subject with request and tracking IDs
            // Remove the REQ- prefix if it already exists in orderNumber to avoid duplication
            const requestRef = (searchRequest.orderNumber || searchRequest.id).toString();
            const cleanRequestRef = requestRef.startsWith('REQ-') ? requestRef : `REQ-${requestRef}`;
            const formattedSubject = `${subject} - [${cleanRequestRef}] [TID:${trackingId}]`;
            
            // Add reference footer to message for tracking
            const referenceFooter = `\n\n!Request Reference: ${cleanRequestRef}\nRequest Tracking ID: ${trackingId}\nPlease include this reference in your reply to ensure proper tracking of your response.`;
            
            const fullMessage = message + referenceFooter;
            
            // Send the email with tracking information and sanitized attachments
            const emailSuccess = await emailService.sendEmail({
              to: supplier.email, 
              subject: formattedSubject, 
              text: fullMessage,
              html: fullMessage.replace(/\n/g, '<br/>'),
              attachments: sanitizedAttachments,
              headers: {
                'X-Tracking-ID': trackingId,
                'X-Request-ID': `REQ-${searchRequest.orderNumber || searchRequest.id}`
              }
            });

            console.log(`Email to ${supplier.email} ${emailSuccess ? 'succeeded' : 'failed'}`);

            return {
              supplier,
              trackingId,
              success: emailSuccess,
            };
          } catch (error) {
            console.error(`Error sending email to ${supplier.email}:`, error);
            return {
              supplier,
              success: false,
              error: String(error)
            };
          }
        })
      );

      // Calculate success stats
      const successCount = emailResults.filter(r => r.success).length;

      if (successCount > 0) {
        // Update request status to "sent" if at least one email was sent successfully
        await storage.updateSearchRequestStatus(searchRequest.id, "sent");
        console.log(`Updated request ${searchRequest.id} status to "sent"`);
      }

      // Log the order number being returned to ensure it's consistent
      console.log(`Email sent successfully. Returning order number: ${searchRequest.orderNumber}`);

      // Добавляем подробный лог об успешности
      console.log(`Email execution summary: ${successCount} of ${selectedSuppliers.length} emails sent successfully.`);
      
      // Формируем детальный ответ со всей информацией, необходимой клиенту
      res.json({ 
        success: successCount > 0, 
        totalCount: selectedSuppliers.length,
        successCount,
        orderNumber: searchRequest.orderNumber,
        requestId: searchRequest.id,
        // Дополнительно фильтруем results, чтобы не передавать слишком много данных
        results: emailResults.map(result => ({
          supplierEmail: result.supplier.email,
          supplierName: result.supplier.name,
          success: result.success,
          trackingId: result.trackingId,
          error: result.error
        }))
      });
    } catch (error) {
      console.error("Email sending error:", error);
      res.status(400).json({ message: "Failed to send emails", error: String(error) });
    }
  });

  // Get all search requests (filtered by user if authenticated)
  app.get("/api/search-requests", async (req, res) => {
    try {
      // If user is authenticated, get only their requests
      const userId = req.session.userId;
      const requests = await storage.getAllSearchRequests(userId);
      res.json(requests);
    } catch (error) {
      console.error("Error fetching search requests:", error);
      res.status(500).json({ message: "Failed to fetch search requests", error: String(error) });
    }
  });

  // Get details of a specific search request
  app.get("/api/search-requests/:id", async (req, res) => {
    try {
      const id = parseInt(req.params.id);
      if (isNaN(id)) {
        return res.status(400).json({ message: "Invalid ID format" });
      }

      const request = await storage.getSearchRequest(id);
      if (!request) {
        return res.status(404).json({ message: `Request with ID ${id} not found` });
      }

      // Get all suppliers contacted for this request
      const requestSuppliers = await storage.getRequestSuppliers(id);

      // Get all responses for this request
      const supplierResponses = await storage.getSupplierResponses(id);

      res.json({
        request,
        requestSuppliers,
        supplierResponses
      });
    } catch (error) {
      console.error(`Error fetching request ${req.params.id}:`, error);
      res.status(500).json({ message: "Failed to fetch request details", error: String(error) });
    }
  });

  // Get suppliers for a specific search request
  app.get("/api/search-requests/:id/suppliers", async (req, res) => {
    try {
      const id = parseInt(req.params.id);
      if (isNaN(id)) {
        return res.status(400).json({ message: "Invalid ID format" });
      }

      const request = await storage.getSearchRequest(id);
      if (!request) {
        return res.status(404).json({ message: `Request with ID ${id} not found` });
      }

      // Get matching suppliers for this request
      const matchedSuppliers = await storage.searchSuppliers(request);

      return res.json({ 
        request: {
          ...request,
          // Convert Date to ISO string to avoid serialization issues
          createdAt: request.createdAt instanceof Date 
            ? request.createdAt.toISOString() 
            : request.createdAt
        }, 
        matchedSuppliers
      });
    } catch (error) {
      console.error("Error getting suppliers for search request:", error);
      return res.status(500).json({ message: "Internal server error", error: String(error) });
    }
  });
  
  // Clone a search request
  app.post("/api/search-requests/:id/clone", async (req, res) => {
    try {
      const id = parseInt(req.params.id);
      if (isNaN(id)) {
        return res.status(400).json({ message: "Invalid ID format" });
      }
      
      // Get the original request
      const originalRequest = await storage.getSearchRequest(id);
      if (!originalRequest) {
        return res.status(404).json({ message: `Request with ID ${id} not found` });
      }
      
      // Create a new request with the same details but a new order number
      const newRequest: InsertSearchRequest = {
        productName: originalRequest.productName,
        productDescription: originalRequest.productDescription,
        quantity: originalRequest.quantity,
        budget: originalRequest.budget,
        timeline: originalRequest.timeline,
        useDbSearch: originalRequest.useDbSearch ?? true,
        useApiSearch: originalRequest.useApiSearch ?? false,
        status: 'active',
        orderNumber: '', // Will be generated by the storage layer
        // userId temporarily disabled until database schema is updated
        // userId: req.session.userId || originalRequest.userId
      };
      
      // Create the new request
      const clonedRequest = await storage.createSearchRequest(newRequest);
      
      console.log(`Cloned request ${originalRequest.id} (${originalRequest.orderNumber}) to new request ${clonedRequest.id} (${clonedRequest.orderNumber})`);
      
      return res.json({
        success: true,
        request: clonedRequest
      });
    } catch (error) {
      console.error(`Error cloning request ${req.params.id}:`, error);
      return res.status(500).json({ 
        success: false,
        message: "Failed to clone request", 
        error: String(error) 
      });
    }
  });

  // Get a search request by order number
  app.get("/api/search-requests/order/:orderNumber", async (req, res) => {
    try {
      const orderNumber = req.params.orderNumber;
      const request = await storage.getSearchRequestByOrderNumber(orderNumber);

      if (!request) {
        return res.status(404).json({ message: `Request with order number ${orderNumber} not found` });
      }

      // Get all suppliers contacted for this request
      const requestSuppliers = await storage.getRequestSuppliers(request.id);

      // Get all responses for this request
      const supplierResponses = await storage.getSupplierResponses(request.id);

      res.json({
        request,
        requestSuppliers,
        supplierResponses
      });
    } catch (error) {
      console.error(`Error fetching request with order number ${req.params.orderNumber}:`, error);
      res.status(500).json({ message: "Failed to fetch request details", error: String(error) });
    }
  });

  // Handle supplier responses (this would typically be called by a webhook or email parser)
  app.post("/api/supplier-responses", async (req, res) => {
    try {
      const responseData = insertSupplierResponseSchema.parse(req.body);
      const supplierResponse = await storage.createSupplierResponse(responseData);

      res.json({
        success: true,
        response: supplierResponse
      });
    } catch (error) {
      console.error("Error saving supplier response:", error);
      res.status(400).json({ message: "Failed to save supplier response", error: String(error) });
    }
  });

  // Get all supplier responses (fallback when no request ID is provided)
  app.get("/api/supplier-responses", async (req, res) => {
    try {
      // Set cache control header to prevent caching
      res.set('Cache-Control', 'no-store');

      const requestIdParam = req.query.requestId;
      
      if (requestIdParam) {
        // If requestId query parameter is provided
        const requestId = parseInt(requestIdParam as string);
        if (isNaN(requestId)) {
          return res.status(400).json({ message: "Invalid request ID format" });
        }
        
        console.log(`Fetching supplier responses for request ID: ${requestId}`);
        const responses = await storage.getSupplierResponses(requestId);
        return res.json(responses);
      } else {
        // For security and performance reasons, we don't want to return all responses
        // Instead, return an empty array since a specific request ID should always be provided
        return res.json([]);
      }
    } catch (error) {
      console.error(`Error in supplier responses:`, error);
      res.status(500).json({ message: "Failed to fetch supplier responses", error: String(error) });
    }
  });

  // Get supplier responses for a specific request ID
  app.get("/api/supplier-responses/:requestId", async (req, res) => {
    try {
      const requestId = parseInt(req.params.requestId);
      if (isNaN(requestId)) {
        return res.status(400).json({ message: "Invalid request ID format" });
      }

      const responses = await storage.getSupplierResponses(requestId);
      res.json(responses);
    } catch (error) {
      console.error(`Error fetching responses for request ID ${req.params.requestId}:`, error);
      res.status(500).json({ message: "Failed to fetch supplier responses", error: String(error) });
    }
  });

  // Webhook endpoint for receiving email replies from suppliers
  app.post("/api/webhooks/email", async (req, res) => {
    try {
      console.log("Received email webhook:", req.body);

      // Extract email data from webhook payload
      // Note: This would need to be adapted to the specific webhook provider's format
      const { 
        from, 
        subject, 
        text, 
        headers = {} 
      } = req.body;

      // Try to extract tracking information from headers or message content
      let trackingId = headers['X-Tracking-ID'] || '';
      let orderNumber = headers['X-Order-Number'] || '';
      let requestId: number | undefined;

      // Helper function to extract tracking info from email content
      const extractTrackingInfo = (content: string): { trackingId?: string; orderNumber?: string } => {
        const result: { trackingId?: string; orderNumber?: string } = {};
        
        // Try to extract tracking ID (typically alphanumeric)
        const trackingMatch = content.match(/tracking\s*(?:id|number|code)?\s*[:#\s]+\s*([a-zA-Z0-9_-]{7,15})/i);
        if (trackingMatch && trackingMatch[1]) {
          result.trackingId = trackingMatch[1].trim();
        }
        
        // Try to extract order number (typically REQ-XXXX-XXXXX format)
        const orderMatch = content.match(/(?:order|request)\s*(?:number|id|#)?\s*[:#\s]+\s*(REQ-\d{4}-\d{5})/i);
        if (orderMatch && orderMatch[1]) {
          result.orderNumber = orderMatch[1].trim();
        }
        
        // Alternative format without REQ prefix
        const altOrderMatch = content.match(/(?:order|request)\s*(?:number|id|#)?\s*[:#\s]+\s*(\d{4}-\d{5})/i);
        if (!result.orderNumber && altOrderMatch && altOrderMatch[1]) {
          result.orderNumber = `REQ-${altOrderMatch[1].trim()}`;
        }
        
        return result;
      }
      
      // If no tracking info in headers, try to extract from email content
      if ((!trackingId || !orderNumber) && text) {
        const extractedInfo = extractTrackingInfo(text);
        trackingId = trackingId || extractedInfo.trackingId || '';
        orderNumber = orderNumber || extractedInfo.orderNumber || '';
      }

      // Try to find the corresponding request and supplier
      let requestSupplier;
      let searchRequest;

      if (trackingId) {
        // Find by tracking ID (most reliable)
        requestSupplier = await storage.getRequestSupplierByTrackingId(trackingId);
        if (requestSupplier) {
          searchRequest = await storage.getSearchRequest(requestSupplier.requestId);
          requestId = requestSupplier.requestId;
        }
      } else if (orderNumber) {
        // Check if the order number has the full REQ prefix, if not, try to add it
        if (!orderNumber.startsWith('REQ-') && orderNumber.includes('-')) {
          // This could be a truncated order number like "-2504-16420"
          // Extract the parts and reconstruct with REQ prefix
          const orderMatch = orderNumber.match(/-(\d{4})-(\d{5})/);
          if (orderMatch && orderMatch[1] && orderMatch[2]) {
            console.log(`Detected truncated order number: ${orderNumber}, reconstructing as REQ-${orderMatch[1]}-${orderMatch[2]}`);
            orderNumber = `REQ-${orderMatch[1]}-${orderMatch[2]}`;
          }
        }

        // Find by order number
        searchRequest = await storage.getSearchRequestByOrderNumber(orderNumber);
        if (searchRequest) {
          console.log(`Found search request by order number: ${orderNumber}`);
          requestId = searchRequest.id;
          // Try to find the supplier by email
          const requestSuppliers = await storage.getRequestSuppliers(searchRequest.id);
          requestSupplier = requestSuppliers.find(s => s.supplierEmail === from);

          if (requestSupplier) {
            console.log(`Found matching supplier by email: ${from}`);
          } else {
            console.log(`No matching supplier found for email: ${from} - will create a generic response record`);
          }
        } else {
          console.log(`No search request found for order number: ${orderNumber}`);
        }
      }

      // For testing/demo purposes, if we still can't find a request, use the latest request
      if (!requestId) {
        console.log("Could not determine which request this email belongs to. Using latest request for testing.");
        const allRequests = await storage.getAllSearchRequests();
        if (allRequests.length > 0) {
          // Sort by createdAt in descending order and get the most recent
          const latestRequest = allRequests.sort((a, b) => {
            const dateA = a.createdAt instanceof Date ? a.createdAt : new Date(a.createdAt || 0);
            const dateB = b.createdAt instanceof Date ? b.createdAt : new Date(b.createdAt || 0);
            return dateB.getTime() - dateA.getTime();
          })[0];

          requestId = latestRequest.id;
          searchRequest = latestRequest;

          // Try to find any supplier for this request
          const requestSuppliers = await storage.getRequestSuppliers(requestId);
          if (requestSuppliers.length > 0) {
            requestSupplier = requestSuppliers[0];
          }
        } else {
          return res.status(400).json({ 
            success: false, 
            message: "Could not determine which request this email belongs to" 
          });
        }
      }

      // Set a default response date if not provided
      const responseDate = new Date();

      // Create a supplier response record
      const supplierResponse = await storage.createSupplierResponse({
        requestId,
        requestSupplierId: requestSupplier?.id,
        supplierId: requestSupplier?.supplierId || 'unknown',
        supplierName: requestSupplier?.supplierName || from,
        supplierEmail: from,
        subject,
        content: text,
        responseDate,
        attachments: req.body.attachments || [],
      });

      res.json({
        success: true,
        message: "Email response recorded successfully",
        response: supplierResponse
      });
    } catch (error) {
      console.error("Error processing email webhook:", error);
      res.status(500).json({ 
        success: false, 
        message: "Failed to process email webhook", 
        error: String(error) 
      });
    }
  });

        });

  // Временное хранилище для контактов групп при переходе между страницами

    contacts: any[];
    groupId: number;
    groupName: string;
    timestamp: number;
  }>();
  
  // Очистка старых данных каждые 5 минут
  setInterval(() => {
    const now = Date.now();
    tempContactsStore.forEach((data, key) => {
      // Удаляем записи старше 15 минут
      if (now - data.timestamp > 15 * 60 * 1000) {
        tempContactsStore.delete(key);
      }
    });

  app.post("/api/temp-contacts", async (req, res) => {
    try {
      const { contacts, groupId, groupName } = req.body;
      
      console.log(`Получен запрос на сохранение контактов во временное хранилище. 
        Группа: ${groupId} (${groupName})
        Контакты: получено ${contacts?.length || 0} элементов
        Данные: ${JSON.stringify(req.body).slice(0, 200)}...`);

        console.error(`Ошибка формата данных: contacts=${typeof contacts}, isArray=${Array.isArray(contacts)}, groupId=${groupId}`);
        return res.status(400).json({ error: "Неверный формат данных" });
      }
      
      // Генерируем уникальный ключ
      const key = `${Date.now()}-${Math.random().toString(36).substring(2, 15)}`;
      
      // Подготавливаем и проверяем данные контактов
      const safeContacts = contacts.map(contact => {
        // Убедимся, что у каждого контакта есть необходимые поля
        return {
          id: contact.id || Math.floor(Math.random() * 10000),
          name: contact.name || 'Контакт',
          email: contact.email || '',
          phone: contact.phone || '',
          // Добавляем другие поля при необходимости
        };
      });
      
      // Сохраняем данные
      tempContactsStore.set(key, {
        contacts: safeContacts,
        groupId,
        groupName,
        timestamp: Date.now()
      });
      
      console.log(`Успешно сохранено ${safeContacts.length} контактов во временное хранилище с ключом ${key}`);
      console.log(`Пример первого контакта: ${JSON.stringify(safeContacts[0])}`);
      
      // Проверяем, что данные действительно сохранились
      const savedData = tempContactsStore.get(key);
      if (savedData) {
        console.log(`Проверка: Данные найдены в хранилище. Контактов: ${savedData.contacts.length}`);
      } else {
        console.warn(`Проверка: Данные НЕ найдены в хранилище после сохранения!`);
      }
      
      res.json({ key });
    } catch (error) {
      console.error("Ошибка при сохранении временных контактов:", error);
      res.status(500).json({ error: "Ошибка при сохранении контактов", message: error.message });
    }
  });
  
  // Получение контактов из временного хранилища
  app.get("/api/temp-contacts/:key", (req, res) => {
    try {
      const { key } = req.params;
      console.log(`Запрос на получение контактов по ключу: ${key}`);
      
      // Получаем все ключи в хранилище для отладки
      const allKeys = Array.from(tempContactsStore.keys());
      console.log(`Доступные ключи в хранилище: ${allKeys.join(', ')}`);
      
      const data = tempContactsStore.get(key);
      
      if (!data) {
        console.warn(`Ключ ${key} не найден в хранилище временных данных`);
        return res.status(404).json({ 
          error: "Данные не найдены или устарели",
          availableKeys: allKeys 
        });
      }
      
      console.log(`Успешно получено ${data.contacts.length} контактов из временного хранилища по ключу ${key}`);
      console.log(`Пример данных: Группа ${data.groupId} (${data.groupName})`);
      
      if (data.contacts && data.contacts.length > 0) {
        console.log(`Пример первого контакта: ${JSON.stringify(data.contacts[0])}`);
      }
      
      // Важно: создаем копию данных перед удалением из хранилища
      const dataCopy = {
        contacts: [...data.contacts],
        groupId: data.groupId,
        groupName: data.groupName,
        timestamp: data.timestamp
      };
      
      // После получения данных удаляем их из хранилища
      tempContactsStore.delete(key);
      console.log(`Данные с ключом ${key} удалены из хранилища`);
      
      res.json(dataCopy);
    } catch (error) {
      console.error("Ошибка при получении временных контактов:", error);
      res.status(500).json({ 
        error: "Ошибка при получении контактов", 
        message: error.message 
      });
    }
  });
