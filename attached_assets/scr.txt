app.post('/api/generate-comparison', async (req, res) => {
  const { supplierIds, parameters } = req.body;
  
  // 1. Получаем реальные данные поставщиков
  const suppliers = await prisma.supplier.findMimport { useState, useEffect } from 'react';
import { useLocation } from 'wouter';
import { 
  Card, 
  CardContent, 
  CardHeader, 
  CardTitle,
  CardDescription,
  CardFooter 
} from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Spinner } from '@/components/ui/spinner';
import { toast } from '@/hooks/use-toast';
import { Download, ArrowLeft, Share2 } from 'lucide-react';
import { apiRequest } from '@/lib/queryClient';
import type { Supplier } from '@shared/schema';

interface Parameter {
  id: string;
  label: string;
  selected: boolean;
}

interface ComparisonData {
  suppliers: string[];
  parameters: string[];
  tableData: {
    parameter: string;
    [supplierId: string]: string | number;
  }[];
  htmlTable: string;
  csv: string;
}

export default function CompareResults() {
  const [, setLocation] = useLocation();
  
  // State
  const [suppliers, setSuppliers] = useState<Supplier[]>([]);
  const [parameters, setParameters] = useState<Parameter[]>([]);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [comparisonData, setComparisonData] = useState<ComparisonData | null>(null);
  
  // Load data from local storage
  useEffect(() => {
    const loadData = async () => {
      try {
        const storedSuppliers = localStorage.getItem('compareSuppliers');
        const storedParameters = localStorage.getItem('compareParameters');
        
        if (!storedSuppliers || !storedParameters) {
          throw new Error("Missing required comparison data");
        }
        
        const parsedSuppliers = JSON.parse(storedSuppliers) as Supplier[];
        const parsedParameters = JSON.parse(storedParameters) as Parameter[];
        
        if (!Array.isArray(parsedSuppliers) || parsedSuppliers.length === 0 ||
            !Array.isArray(parsedParameters) || parsedParameters.length === 0) {
          throw new Error("Invalid comparison data format");
        }
        
        setSuppliers(parsedSuppliers);
        setParameters(parsedParameters);
        
        // Generate comparison with real data
        await generateComparison(parsedSuppliers, parsedParameters);
        
      } catch (error) {
        console.error("Error loading comparison data:", error);
        setError(error instanceof Error ? error.message : "Failed to load comparison data");
        setIsLoading(false);
        
        toast({
          title: "Error",
          description: "Failed to load comparison data. Please try again.",
          variant: "destructive"
        });
        
        setTimeout(() => setLocation('/dashboard'), 2000);
      }
    };

    loadData();
  }, [setLocation]);
  
  // Generate comparison with real data
  const generateComparison = async (suppliers: Supplier[], parameters: Parameter[]) => {
    try {
      setIsLoading(true);
      
      // Prepare request data
      const requestData = {
        supplierIds: suppliers.map(s => s.id),
        parameters: parameters.filter(p => p.selected).map(p => p.id)
      };
      
      // Call API to get real comparison data
      const response = await apiRequest<{
        success: boolean;
        data: ComparisonData;
      }>('/api/generate-comparison', 'POST', requestData);
      
      if (!response?.success) {
        throw new Error("Failed to generate comparison");
      }
      
      setComparisonData(response.data);
      
    } catch (error) {
      console.error("Error generating comparison:", error);
      setError("Failed to generate comparison. Please try again.");
      
      toast({
        title: "Error",
        description: error instanceof Error ? error.message : "Comparison failed",
        variant: "destructive"
      });
    } finally {
      setIsLoading(false);
    }
  };
  
  // Download CSV file
  const downloadCSV = () => {
    if (!comparisonData) return;
    
    const blob = new Blob([comparisonData.csv], { type: 'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const link = document.createElement('a');
    link.setAttribute('href', url);
    link.setAttribute('download', `suppliers-comparison-${new Date().toISOString().slice(0, 10)}.csv`);
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
  };
  
  // Share comparison
  const shareComparison = () => {
    navigator.clipboard.writeText(window.location.href);
    toast({
      title: "Link copied",
      description: "Share this link to view the comparison"
    });
  };

  if (isLoading) {
    return (
      <div className="container max-w-4xl py-8">
        <Card>
          <CardContent className="pt-8 pb-8 flex flex-col items-center justify-center">
            <Spinner className="h-8 w-8 mb-4" />
            <p className="text-center text-lg font-medium">Generating comparison report...</p>
            <p className="text-center text-muted-foreground mt-2">
              Comparing {suppliers.length} suppliers by {parameters.filter(p => p.selected).length} parameters
            </p>
          </CardContent>
        </Card>
      </div>
    );
  }
  
  if (error) {
    return (
      <div className="container max-w-4xl py-8">
        <Card>
          <CardContent className="pt-8 pb-8">
            <p className="text-center text-red-500">{error}</p>
            <div className="flex justify-center mt-4">
              <Button onClick={() => setLocation('/dashboard')}>
                Back to Dashboard
              </Button>
            </div>
          </CardContent>
        </Card>
      </div>
    );
  }
  
  return (
    <div className="container max-w-5xl py-8">
      <Card className="mb-8">
        <CardHeader>
          <div className="flex items-center justify-between">
            <CardTitle className="text-2xl">Supplier Comparison</CardTitle>
            <div className="flex items-center gap-2">
              <Button 
                variant="outline" 
                className="gap-1" 
                onClick={shareComparison}
              >
                <Share2 className="h-4 w-4" />
                Share
              </Button>
              <Button 
                className="gap-1" 
                onClick={downloadCSV}
                disabled={!comparisonData}
              >
                <Download className="h-4 w-4" />
                Download CSV
              </Button>
            </div>
          </div>
          <CardDescription>
            {comparisonData 
              ? `Comparing ${comparisonData.suppliers.length} suppliers by ${comparisonData.parameters.length} criteria`
              : 'No comparison data available'}
          </CardDescription>
        </CardHeader>
        
        <CardContent>
          {comparisonData ? (
            <div className="space-y-6">
              {comparisonData.htmlTable ? (
                <div 
                  className="comparison-table overflow-x-auto border rounded-lg"
                  dangerouslySetInnerHTML={{ __html: comparisonData.htmlTable }}
                />
              ) : (
                <div className="grid gap-4">
                  {comparisonData.tableData.map((row, index) => (
                    <div key={index} className="grid grid-cols-12 gap-4 items-center border-b py-2">
                      <div className="col-span-3 font-medium">{row.parameter}</div>
                      {suppliers.map(supplier => (
                        <div key={supplier.id} className="col-span-3">
                          {row[supplier.id] || 'N/A'}
                        </div>
                      ))}
                    </div>
                  ))}
                </div>
              )}
            </div>
          ) : (
            <p className="text-center text-muted-foreground py-8">
              No comparison data available
            </p>
          )}
        </CardContent>
        
        <CardFooter>
          <Button 
            variant="outline" 
            className="gap-1"
            onClick={() => setLocation('/compare-parameters')}
          >
            <ArrowLeft className="h-4 w-4" />
            Change Parameters
          </Button>
        </CardFooter>
      </Card>
    </div>
  );
}any({
    where: { id: { in: supplierIds } },
    include: {
      products: true,
      certifications: true,
      pricingHistory: {
        orderBy: { date: 'desc' },
        take: 1
      }
    }
  });

  // 2. Формируем таблицу сравнения
  const comparisonTable = parameters.map(param => {
    const row = { parameter: param };
    
    suppliers.forEach(supplier => {
      // Получаем значение параметра для каждого поставщика
      row[supplier.id] = getSupplierParamValue(supplier, param);
    });
    
    return row;
  });

  // 3. Генерируем CSV
  const csv = generateCSV(comparisonTable, suppliers);

  // 4. Возвращаем структурированные данные
  res.json({
    success: true,
    data: {
      suppliers: suppliers.map(s => s.name),
      parameters,
      tableData: comparisonTable,
      htmlTable: generateHTMLTable(comparisonTable),
      csv
    }
  });
});

// Вспомогательная функция для получения значений параметров
function getSupplierParamValue(supplier, param) {
  switch(param) {
    case 'price':
      return supplier.pricingHistory?.[0]?.price || 'N/A';
    case 'delivery_time':
      return supplier.avgDeliveryDays + ' days';
    case 'certifications':
      return supplier.certifications.map(c => c.name).join(', ');
    // ... другие параметры
    default:
      return supplier[param] || 'N/A';
  }
}
