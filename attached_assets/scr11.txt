// Удалить старую функцию extractParameterValue и заменить на:

interface ExtractionResult {
value: string;
source: 'content' | 'attachment';
confidence: number;
}

async function extractParameterValue(
content: string,
attachments: any[],
parameter: string
): Promise<ExtractionResult> {
const extractionMethods = [
extractFromStructuredTable,
extractFromKeyValuePairs,
extractUsingNLP,
extractFromAttachments
];

let bestResult: ExtractionResult = { value: '-', source: 'content', confidence: 0 };

for (const method of extractionMethods) {
const result = await method(content, attachments, parameter);
if (result.confidence > bestResult.confidence) {
bestResult = result;
}
}

return bestResult;
}

async function extractFromStructuredTable(
content: string,
attachments: any[],
parameter: string
): Promise<ExtractionResult> {
// Улучшенный парсинг HTML/XML таблиц
const tableRegex = /<table[^>]>[\s\S]?</table>/gi;
const tables = content.match(tableRegex) || [];

for (const table of tables) {
const rows = table.match(/<tr[^>]>[\s\S]?</tr>/gi) || [];
for (const row of rows) {
const columns = row.match(/<t[hd][^>]>[\s\S]?</t[hd]>/gi) || [];
if (columns.length >= 2) {
const paramCol = columns[0].replace(/<[^>]+>/g, '').trim();
if (similarity(paramCol, parameter) > 0.8) {
const value = columns[1].replace(/<[^>]+>/g, '').trim();
return { value, source: 'content', confidence: 0.9 };
}
}
}
}
return { value: '-', source: 'content', confidence: 0 };
}
