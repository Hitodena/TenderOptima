// Добавить новый класс для генерации отчетов
class ReportGenerator {
  private data: any[];
  private headers: string[];
  private requestDetails: any;

  constructor(data: any[], headers: string[], requestDetails?: any) {
    this.data = data;
    this.headers = headers;
    this.requestDetails = requestDetails;
  }

  generateExcel(): Buffer {
    const workbook = new ExcelJS.Workbook();
    const worksheet = workbook.addWorksheet('Supplier Comparison');
    
    // Форматирование заголовков
    worksheet.columns = this.headers.map(header => ({
      header: header.toUpperCase(),
      key: header,
      width: 25
    }));

    // Добавление данных
    this.data.forEach(row => {
      worksheet.addRow(row);
    });

    // Стилизация
    worksheet.eachRow({ includeEmpty: true }, (row, rowNumber) => {
      row.eachCell(cell => {
        cell.border = {
          top: { style: 'thin' },
          left: { style: 'thin' },
          bottom: { style: 'thin' },
          right: { style: 'thin' }
        };
      });
    });

    return workbook.xlsx.writeBuffer();
  }

  generatePDF(): Promise<Buffer> {
    return new Promise((resolve, reject) => {
      const doc = new PDFDocument();
      const buffers: any[] = [];
      
      doc.on('data', buffers.push.bind(buffers));
      doc.on('end', () => resolve(Buffer.concat(buffers)));
      
      // Заголовок отчета
      doc.fontSize(18).text('Supplier Comparison Report', { align: 'center' });
      doc.moveDown();
      
      // Таблица с данными
      const table = {
        headers: this.headers,
        rows: this.data.map(row => this.headers.map(h => row[h]))
      };
      
      doc.table(table, {
        prepareHeader: () => doc.font('Helvetica-Bold'),
        prepareRow: (row, i) => doc.font('Helvetica').fontSize(12)
      });
      
      doc.end();
    });
  }
}
