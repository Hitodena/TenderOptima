```bash

LOG_LEVEL=info
LOG_DIR=./logs
MONITORING_ENABLED=true
ALERT_THRESHOLDS_ENABLED=true
METRICS_ENABLED=true
DASHBOARD_ENABLED=true
```

### Зависимости
```json
{
  "dependencies": {
    "express": "^4.18.2",
    "date-fns": "^2.30.0",
    "fs": "^0.0.1-security"
  }
}
```

### Конфигурация мониторинга
```typescript
// monitoring.config.ts
export const monitoringConfig = {
  health: {
    enabled: true,
    interval: 30000, // 30 секунд
    thresholds: {
      memory: 90,
      cpu: 80,
      responseTime: 5000
    }
  },
  logging: {
    level: 'info',
    directory: './logs',
    maxFiles: 30,
    maxSize: '10MB'
  },
  alerts: {
    enabled: true,
    thresholds: {
      memory_usage: 90,
      cpu_usage: 80,
      response_time: 5000,
      error_rate: 10
    }
  },
  metrics: {
    enabled: true,
    retention: 1000,
    export: {
      prometheus: true,
      json: true
    }
  },
  dashboard: {
    enabled: true,
    refreshInterval: 5000,
    historicalData: true
  }
};
```
